    <script type="py">
        import random
        from pyscript import document

        # --- DATA ---
        PLACES = [
            "the empty lot with no motel",
            "I’m only making you do it — it’s exactly what you wants",
            "the wind corridor",
            "the autoclave room",
            "don’t say that. don’t open your mouth",
            "the smoke-layered roof edge",
            "the dog’s ‘humanized moan’\naegoo aagoo aahfogoo\naggegoogo aggogoo agogosoooo",
            "the industrial field of soft black plants",
            "the canyon of metal fragments",
            "can you shut the fuck up"
        ]

        DESCRIPTIONS = [
            "you open my mouth, then “sell-speak” through me",
            "Rectangles repeat themselves in quiet patterns.",
            "A soft mechanical hum drifts through the air.",
            "I’m afraid of being left.",
            "“I really hate women,” you chant, and you mean it",
            "The wind carries small particles that shimmer briefly.",
            "praying.",
            "The ground feels like a document someone forgot to finish.",
            "that’s what you want.",
            "You feel watched, but only by the architecture."
        ]

        EVENTS = [
            "A rectangle shifts slightly, then stops.",
            "A distant machine coughs and resets.",
            "A soft pulse of light moves across the ground.",
            "Something overhead flickers, then resolves into nothing.",
            "A drifting piece of dust seems to follow you.",
            "A pattern repeats, then repeats differently.",
            "A quiet tone rises, then dissolves.",
            "The air thickens, then clears.",
            "A structure rearranges itself when you look away."
        ]

        # --- SETUP ---
        size = 5
        town = {}
        for x in range(size):
            for y in range(size):
                town[(x, y)] = {
                    "place": random.choice(PLACES),
                    "description": random.choice(DESCRIPTIONS)
                }

        p_x = 2
        p_y = 2
        history = []

        def log(text):
            history.append(text)
            if len(history) > 10: history.pop(0) # Keep terminal clean
            document.querySelector("#output").innerText = "\n\n".join(history)
            # Auto-scroll
            out_div = document.querySelector("#output")
            out_div.scrollTop = out_div.scrollHeight

        def describe_location():
            loc = town[(p_x, p_y)]
            msg = f"You arrive at {loc['place']}.\n{loc['description']}\n{random.choice(EVENTS)}"
            log(msg)

        def move(direction):
            global p_x, p_y
            dx, dy = 0, 0
            if direction == "north": dy = -1
            elif direction == "south": dy = 1
            elif direction == "east": dx = 1
            elif direction == "west": dx = -1

            nx, ny = p_x + dx, p_y + dy

            if (nx, ny) in town:
                p_x, p_y = nx, ny
                describe_location()
            else:
                log("The land becomes abstract and refuses to let you pass.")

        # INITIAL START
        history.append("You enter a town with no motels, no hotels, no agencies.\nA place that feels translated from a language that never fully existed.")
        describe_location()
    </script>
